.NAME
    xNetAdapterRDMA

# Description
    
    This resource is used to enable or disable RDMA on a network adapter.
    
.PARAMETER Name
    Key - String
    Specifies the name of network adapter for which RDMA needs to be configured.

.PARAMETER Enabled
    Write - Boolean
    Specifies whether RDMA is enabled or disabled.


    .EXAMPLE
    This configuration disables RDMA setting on the network adapter.

Configuration Example
{
    param
    (
        [Parameter()]
        [System.String[]]
        $NodeName = 'localhost'
    )

    Import-DSCResource -ModuleName xNetworking

    Node $NodeName
    {
        xNetAdapterRDMA SMBAdapter1
        {
            Name = 'SMB1_1'
            Enabled = $false
        }
    }
}


    .EXAMPLE
    This configuration enables RDMA setting on the network adapter.

Configuration Example
{
    param
    (
        [Parameter()]
        [System.String[]]
        $NodeName = 'localhost'
    )

    Import-DSCResource -ModuleName xNetworking

    Node $NodeName
    {
        xNetAdapterRDMA SMBAdapter1
        {
            Name = 'SMB1_1'
            Enabled = $true
        }
    }
}
